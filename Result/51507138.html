    <?php 
    session_start(); 
    $ip = $_SERVER["REMOTE_ADDR"]; 
    $logoyse = 0; 
    if (!isset($_SESSION["logoid"])) 
    { 
        $_SESSION["logoid"] = $ip; 
        $logoyse = 1; 
    } 
    if ($logoyse == 1) 
    { 
        echo("您是第一次进入！"); 
    } 
    else
    { 
        echo("您不是第一次进入！"); 
    } 
    ?>
